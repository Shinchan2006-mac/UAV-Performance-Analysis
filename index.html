% Project Assignment 4: Trim, Thrust, and Power Analysis
% UAV Configuration: Cropped Delta Wing

clear; clc;close all;

--- Given Data & Constants ---

m = 3.5;            % UAV Mass (kg)
g = 10;             % Gravity (m/s^2)
rho = 1.225;        % Air density (kg/m^3)
W = m * g;          % Weight (N)

% Geometry
Cr = 0.9;           % Root chord (m)
Ct = 0.15;          % Tip chord (m)
b = 1.5;            % Span (m)
e = 0.89;           % Oswald efficiency factor

% Aerodynamic Coefficients
CD0 = 0.03;
Cm0 = 0.01;
CLa = 2.92;         % per rad
CLde = 0.265;       % CL_delta_e
Cma = -0.292;       % Cm_alpha
Cmde = -0.4;        % Cm_delta_e

--- Preprocessing Calculations ---

lambda = Ct / Cr;               % Taper Ratio
S = (b/2) * Cr * (1 + lambda);  % Planform Area (m^2)
AR = b^2 / S;                   % Aspect Ratio
k = 1 / (pi * e * AR);          % Induced Drag Correction Factor

--- Phase 1: Iterative Solver ---

Define Angle of Attack range (0 to 12 degrees in steps of 0.5)

alpha_deg = 0:0.5:12;
alpha_rad = deg2rad(alpha_deg); % Convert to radians for calculations

% Pre-allocate arrays for efficiency
de_trim = zeros(size(alpha_rad));
CL_trim = zeros(size(alpha_rad));
V_trim  = zeros(size(alpha_rad));
CD      = zeros(size(alpha_rad));
Tr      = zeros(size(alpha_rad));
Pr      = zeros(size(alpha_rad));
LD_ratio = zeros(size(alpha_rad));
CL32_CD  = zeros(size(alpha_rad));

for i = 1:length(alpha_rad)

    a = alpha_rad(i);

    % 1. Compute Trim Elevator Deflection (de)
    % From Cm = Cm0 + Cma*a + Cmde*de = 0
    de_trim(i) = -(Cm0 + Cma * a) / Cmde;

    % 2. Compute Trim Lift Coefficient (CL)
    CL_trim(i) = CLa * a + CLde * de_trim(i);

    % 3. Determine Flight Velocity (V)
    % From L = W = 0.5 * rho * V^2 * S * CL
    if CL_trim(i) > 0
        V_trim(i) = sqrt((2 * W) / (rho * S * CL_trim(i)));
    else
        V_trim(i) = NaN; % Avoid division by zero/negative lift
    end

    % 4. Compute Drag Coefficient (CD)
    CD(i) = CD0 + k * CL_trim(i)^2;

    % 5. Calculate Thrust Required (Tr = Drag)
    Tr(i) = 0.5 * rho * V_trim(i)^2 * S * CD(i);

    % 6. Calculate Power Required (Pr = Tr * V)
    Pr(i) = Tr(i) * V_trim(i);

--- Phase 2: Performance Metrics ---

    LD_ratio(i) = CL_trim(i) / CD(i);
    CL32_CD(i) = (CL_trim(i)^(1.5)) / CD(i);

end

--- Results Table ---

Combine results into a table for viewing

--- Phase 3: Visualization ---

figure('Name', 'UAV Performance Analysis');

% 1. Trim Angle of Attack vs Velocity
% This shows the alpha required to maintain level flight at a given speed
subplot(4,2,1);
plot(V_trim, alpha_deg, 'b', 'LineWidth', 2);
grid on;
xlabel('Velocity (m/s)');
ylabel('Trim \alpha (deg)');
title('Trim Angle of Attack vs Velocity');

% 2. Trim Elevator Deflection vs Velocity
subplot(4,2,2);
plot(V_trim, rad2deg(de_trim), 'LineWidth', 1.5);
grid on;
xlabel('Velocity (m/s)');
ylabel('\delta_{e,trim} (deg)');
title('Trim Elevator Deflection');

% 3. Lift Coefficient vs Velocity
subplot(4,2,3);
plot(V_trim, CL_trim, 'LineWidth', 1.5);
grid on;
xlabel('Velocity (m/s)');
ylabel('C_L');
title('Lift Coefficient');

% 4. Drag Coefficient vs Velocity
subplot(4,2,4);
plot(V_trim, CD, 'LineWidth', 1.5);
grid on;
xlabel('Velocity (m/s)');
ylabel('C_D');
title('Drag Coefficient');

% 5. Thrust Required vs Velocity
subplot(4,2,5);
plot(V_trim, Tr, 'r', 'LineWidth', 1.5);
grid on;
xlabel('Velocity (m/s)');
ylabel('Thrust Required (N)');
title('Thrust Required (Drag)');

% 6. Power Required vs Velocity
subplot(4,2,6);
plot(V_trim, Pr, 'g', 'LineWidth', 1.5);
grid on;
xlabel('Velocity (m/s)');
ylabel('Power Required (W)');
title('Power Required');

% 7. Efficiency Metrics (L/D)
subplot(4,2,7);

xlabel('Velocity (m/s)');
ylabel('C_L / C_D');
title('Aerodynamic Efficiency (L/D)');

% 8. Power Factor (Endurance Metric) vs Velocity

xlabel('Velocity (m/s)');
ylabel('C_L^{3/2} / C_D');
title('Endurance Metric (C_L^{1.5} / C_D)');
